    int mid = 5;
    for (int i = 0; i < mid; i++) {
        for (int j = 1; j <= 9; j++) {
            int lo = mid - i;
            int hi = mid + i;
            if (j >= lo || j <= hi) {
                cout << '*';
            } else {
                cout << ' ';
            }
            cout << i;
        }
        cout << '\n';
    }

    return 0;
}
